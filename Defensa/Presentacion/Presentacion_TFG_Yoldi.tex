%-------------------------------------------------
\begin{frame}{Distorsión óptica}
\begin{itemize}
    \item \textbf{Qué es:} deformación geométrica variable con la distancia al centro óptico.
    \item \textbf{Tipos:}
    \begin{itemize}
        \item \textbf{Barrel}: líneas rectas se curvan hacia fuera.
        \item \textbf{Pincushion}: líneas rectas se curvan hacia dentro.
    \end{itemize}
    \item \textbf{Impacto:} errores de medida/registro y georreferenciación.
    \item \textbf{Corrección:} modelado matemático $\rightarrow$ cuantificar $\rightarrow$ remapear (corregir la imagen).
\end{itemize}
\end{frame}

\section{Modelado matemático de la distorsión}
%-------------------------------------------------
\begin{frame}{Modelos clásicos de distorsión}

\small
\begin{columns}[T,onlytextwidth]

\column{0.5\textwidth}
\centering
\textbf{Modelo polinómico (Duane)}
\vspace{0.2cm}
\[
\left\{
\begin{aligned}
x &= \left(1 + k_1 \|P\|^2 + k_2 \|P\|^4 + \dots \right) u \\
y &= \left(1 + k_1 \|P\|^2 + k_2 \|P\|^4 + \dots \right) v
\end{aligned}
\right.
\]

\vspace{0.45cm}

\scriptsize
\(k_1, k_2, \dots\): parámetros de distorsión.

\column{0.5\textwidth}
\centering
\textbf{Modelo de división (Fitzgibbon)}
\[
\left\{
\begin{aligned}
u &= \dfrac{x}{1 + \lambda_1 \|X\|^2 + \lambda_2 \|X\|^4 + \dots} \\
v &= \dfrac{y}{1 + \lambda_1 \|X\|^2 + \lambda_2 \|X\|^4 + \dots}
\end{aligned}
\right.
\]

\scriptsize
\(\lambda_1, \lambda_2, \dots\): parámetros de distorsión.

\end{columns}

\vspace{0.3cm}
{\footnotesize
\(P=(u,v)\): coordenadas ideales, sin distorsión, relativas al centro de distorsión del sistema.\\
\(X=(x,y)\): coordenadas distorsionadas relativas al centro de distorsión del sistema.\\
}
\vspace{0.4cm}
\begin{itemize}
    \item Más coeficientes $\Rightarrow$ más precisión; para cámaras profesionales suele bastar 1 coeficiente.
\end{itemize}
\end{frame}

%-------------------------------------------------
\begin{frame}{Modelo utilizado}
Modelo radial con un coeficiente:
\begin{equation*}
    R' = (1 + kR^2)R
\end{equation*}

\begin{itemize}
    \item $R$ : posición ideal
    \item $R'$ : posición distorsionada
    \item $k$ : coeficiente de distorsión
\end{itemize}

\begin{itemize}
    \item $k > 0$ $\rightarrow$ pincushion
    \item $k < 0$ $\rightarrow$ barrel
\end{itemize}
\end{frame}

\begin{frame}{5) Cálculo del centroide ponderado}
\begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
        {\scriptsize
        \begin{equation*}
        M_{i_x} =
        \frac{
        \displaystyle \sum_{j=1}^{U} \sum_{k=1}^{V} \left( I(j,k) - I_{umbral} \right) H(j,k)\, x_{jk}
        }{
        \displaystyle \sum_{j=1}^{U} \sum_{k=1}^{V} \left( I(j,k) - I_{umbral} \right) H(j,k)
        }
        \end{equation*}
        \vspace{-0.6em}
        \begin{equation*}
        M_{i_y} =
        \frac{
        \displaystyle \sum_{j=1}^{U} \sum_{k=1}^{V} \left( I(j,k) - I_{umbral} \right) H(j,k)\, y_{jk}
        }{
        \displaystyle \sum_{j=1}^{U} \sum_{k=1}^{V} \left( I(j,k) - I_{umbral} \right) H(j,k)
        }
        \end{equation*}
        \vspace{-0.6em}
        \begin{equation*}
        H(j, k) =
        \begin{cases}
        1, & \text{si } I(j,k) \geq I_{umbral} \\
        0, & \text{si } I(j,k) < I_{umbral}
        \end{cases}
        \end{equation*}
        }
    \begin{itemize}
        \item Precisión subpíxel.
    \end{itemize}

    \column{0.52\textwidth}
    \centering
    \includegraphics[width=0.75\linewidth]{Imagenes/Centroide_ventana_binaria.png}\\
\end{columns}

\begin{frame}{Ajuste por mínimos cuadrados}

\small
\begin{columns}[T,onlytextwidth]

\column{0.5\textwidth}

Modelo radial:

\[
R' = (1 + kR^2)R
\]

Reordenando:

\[
R' - R = kR^3
\]

\vspace{0.3cm}

\[
R_i = \|M_i\|, 
\quad 
R'_i = \|P_i\|
\]

{\footnotesize
$M_i$: punto ideal \\
$P_i$: punto distorsionado
}

\column{0.5\textwidth}

Relación lineal:

\[
y_i = x_i\, k
\]

\[
y_i = R'_i - R_i
\]

\[
x_i = R_i^3
\]

\vspace{0.5cm}

\centering
\textbf{Estimación de $k$ \\ mediante regresión lineal}

\end{columns}

\end{frame}
