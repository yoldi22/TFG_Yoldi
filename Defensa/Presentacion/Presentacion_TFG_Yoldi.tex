\section{Contexto y motivación}
%-------------------------------------------------
\begin{frame}{Contexto y motivación}
\begin{itemize}
    \item Crecimiento de la observación terrestre mediante microsatélites
    \item Cámaras de alta resolución espacial
    \item Necesidad de precisión geométrica
    \item \lightit{Image Processing}
\end{itemize}
\end{frame}

%-------------------------------------------------
\begin{frame}{Degradaciones en sistemas ópticos}

\begin{columns}[T,onlytextwidth]

\column{0.48\textwidth}

\textbf{Principales fuentes de degradación}

\begin{itemize}
    \item Ruido electrónico
    \begin{itemize}
        \item Lectura
        \item Disparo
        \item Corriente oscura
    \end{itemize}
    \item Aberraciones ópticas
    \begin{itemize}
        \item Esférica, coma, astigmatismo
        \item Curvatura de campo
        \item \alert{\textbf{Distorsión}}
    \end{itemize}
\end{itemize}

\vspace{0.3cm}

\alert{En este trabajo nos centramos en la distorsión geométrica.}

\column{0.52\textwidth}

\textbf{Distorsión óptica}

\vspace{0.3cm}

Deformación geométrica que aumenta con la distancia al centro óptico.

\vspace{0.4cm}

\begin{itemize}
    \item \textbf{Barrel}: líneas se curvan hacia fuera
    \item \textbf{Pincushion}: líneas se curvan hacia dentro
\end{itemize}

\vspace{0.4cm}

Impacto: errores en medida y georreferenciación.

\end{columns}

\end{frame}

\section{Modelado matemático de la distorsión}
%-------------------------------------------------
\begin{frame}{Modelos clásicos de distorsión}

\small
\begin{columns}[T,onlytextwidth]

\column{0.5\textwidth}
\centering
\textbf{Modelo polinómico (Duane)}
\vspace{0.3cm}
\[
R_d = (1 + k_1 R^2 + k_2 R^4 + \dots)\ R
\]


\column{0.5\textwidth}
\centering
\textbf{Modelo de división (Fitzgibbon)}
\vspace{0.25cm}
\[
R = \dfrac{R_d}{1 + \lambda_1 R_d^2 + \lambda_2 R_d^4 + \dots}
\]

\end{columns}
\vspace{0.4cm}
\scriptsize
$R$: posición ideal, sin distorisón\\
$R_d$: posición real, distorsionada
\vspace{0.6cm}

\begin{itemize}
    \item Ambos modelos utilizan expansiones radiales.
    \item Más coeficientes $\Rightarrow$ mayor precisión.
    \item \alert{\textbf{En sistemas ópticos profesionales suele ser suficiente un único coeficiente.}}
\end{itemize}

\end{frame}
%-------------------------------------------------
\begin{frame}{Modelo utilizado}
Modelo radial con un coeficiente:
\begin{equation*}
    R_d = (1 + kR^2)R
\end{equation*}

\begin{itemize}
    \item $R$ : posición ideal
    \item $R_d$ : posición distorsionada
    \item $k$ : coeficiente de distorsión
\end{itemize}

\begin{itemize}
    \item $k > 0$ $\rightarrow$ pincushion
    \item $k < 0$ $\rightarrow$ barrel
\end{itemize}
\end{frame}

\begin{frame}{Ajuste por mínimos cuadrados}

\begin{columns}[T,onlytextwidth]

\column{0.48\textwidth}

\small
\textbf{Modelo radial}

\vspace{0.2cm}

\[
R_d = (1 + kR^2)R
\]

\vspace{0.4cm}

\textbf{Reordenación}

\[
R_d - R = kR^3
\]

\column{0.52\textwidth}

\small
\textbf{\alert{Linealización}}

\vspace{0.3cm}

\[
y_i = x_i\, k
\]

\vspace{0.3cm}

\[
y = R_d - R
\quad
x = R^3
\]

\vspace{0.6cm}

\centering
\alert{\textbf{Estimación de $k$ mediante ajuste lineal}}

\vspace{0.4cm}

\begin{itemize}
    \item Eliminación de outliers con Z-Score
\end{itemize}
